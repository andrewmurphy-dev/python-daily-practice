from pydantic import BaseModel, Field


class Contact(BaseModel):
    name: str = Field(min_length=1)
    email: str



#you do not write 

def __init__(self, name, email):
    self.name = name
    self.email = email


#why?

BaseModel already handles that for you ! 

pydantic creates the __init__ nehavior behind the scenes
